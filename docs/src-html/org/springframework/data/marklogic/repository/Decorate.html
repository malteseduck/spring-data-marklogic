001package org.springframework.data.marklogic.repository;
002
003import org.springframework.data.annotation.QueryAnnotation;
004import org.springframework.data.marklogic.repository.query.QueryDecorator;
005
006import java.lang.annotation.*;
007
008/**
009 * Allows application of a query "decorator" that can add common query logic to all the structured queries being sent to
010 * the database, i.e. common security logic, transforms with parameters, collections, etc.
011 */
012@Retention(RetentionPolicy.RUNTIME)
013@Target({ ElementType.METHOD, ElementType.ANNOTATION_TYPE })
014@Documented
015@QueryAnnotation
016public @interface Decorate {
017
018    /**
019     * Allows the specification of a decorator class that will add a set piece of query logic to the StructuredQueryDefinition
020     * being created by the finder query to which this is attached.
021     *
022     * @return
023     */
024    Class<? extends QueryDecorator> using();
025}
