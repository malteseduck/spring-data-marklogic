016package org.springframework.data.marklogic.repository.support;
017
018import org.springframework.data.mapping.context.MappingContext;
019import org.springframework.data.marklogic.core.MarkLogicOperations;
020import org.springframework.data.repository.Repository;
021import org.springframework.data.repository.core.support.RepositoryFactorySupport;
022import org.springframework.data.repository.core.support.TransactionalRepositoryFactoryBeanSupport;
023import org.springframework.util.Assert;
024
025import java.io.Serializable;
026
027public class MarkLogicRepositoryFactoryBean<T extends Repository<S, ID>, S, ID extends Serializable>
028        extends TransactionalRepositoryFactoryBeanSupport<T, S, ID> {
029
030    private MarkLogicOperations operations;
031    private boolean mappingContextConfigured = false;
032
033    public MarkLogicRepositoryFactoryBean(Class<? extends T> repositoryInterface) {
034        super(repositoryInterface);
035    }
036
037    public void setMarkLogicOperations(MarkLogicOperations operations) {
038        this.operations = operations;
039    }
040
041    /*
042     * (non-Javadoc)
043     * @see org.springframework.data.repository.core.support.RepositoryFactoryBeanSupport#setMappingContext(org.springframework.data.mapping.context.MappingContext)
044     */
045    @Override
046    protected void setMappingContext(MappingContext<?, ?> mappingContext) {
047        super.setMappingContext(mappingContext);
048        this.mappingContextConfigured = true;
049    }
050
051    @Override
052    protected RepositoryFactorySupport doCreateRepositoryFactory() {
053        return getFactoryInstance(operations);
054    }
055
056    protected RepositoryFactorySupport getFactoryInstance(MarkLogicOperations operations) {
057        return new MarkLogicRepositoryFactory(operations);
058    }
059
060    @Override
061    public void setTransactionManager(String transactionManager) {
062        super.setTransactionManager(transactionManager);
063    }
064
065    /*
066         * (non-Javadoc)
067         *
068         * @see
069         * org.springframework.data.repository.support.RepositoryFactoryBeanSupport
070         * #afterPropertiesSet()
071         */
072    @Override
073    public void afterPropertiesSet() {
074
075        super.afterPropertiesSet();
076        Assert.notNull(operations, "MarkLogicTemplate must not be null!");
077
078        if (!mappingContextConfigured) {
079            setMappingContext(operations.getConverter().getMappingContext());
080        }
081    }
082}
