001package org.springframework.data.marklogic.repository.support;
002
003import org.springframework.data.marklogic.core.mapping.MarkLogicPersistentEntity;
004import org.springframework.data.marklogic.repository.query.MarkLogicEntityInformation;
005import org.springframework.data.repository.core.support.PersistentEntityInformation;
006
007import java.io.Serializable;
008
009public class MappingMarkLogicEntityInformation <T, ID extends Serializable> extends PersistentEntityInformation<T, ID>
010        implements MarkLogicEntityInformation<T, ID> {
011
012    private final MarkLogicPersistentEntity<T> entityMetadata;
013    private final Class<ID> fallbackIdType;
014
015    public MappingMarkLogicEntityInformation(MarkLogicPersistentEntity<T> entityMetadata) {
016        this(entityMetadata, null);
017    }
018
019    public MappingMarkLogicEntityInformation(MarkLogicPersistentEntity<T> entityMetadata, Class<ID> idType) {
020        super(entityMetadata);
021        this.entityMetadata = entityMetadata;
022        this.fallbackIdType =  idType;
023    }
024
025
026    @Override
027    public String getCollectionName() {
028        return entityMetadata.getCollection();
029    }
030
031    @Override
032    public String getIdAttribute() {
033        return entityMetadata.getIdProperty().getName();
034    }
035
036    @Override
037    public Class<ID> getIdType() {
038        if (this.entityMetadata.hasIdProperty()) {
039            return super.getIdType();
040        }
041        throw new IllegalArgumentException("Entity must be annotated with @Id");
042    }
043}
